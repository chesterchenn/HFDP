interface Subject {
  registerObserver()
  removeObserver()
  notifyObserver()
}

class WeatherData {
  registerObserver()
  removeObserver()
  notifyObserver()

  getTemperature()
  getHumidity()
  getPressure()
  measurementsChanged()
}

interface Observer {
  update()
}

interface DisplayElement {
  display()
}

class CurrentConditionsDisplay {
  update()
  display()
}

class StatisticsDisplay {
  update()
  display()
}

class ForecastDisplay {
  update()
  display()
}

class ThirdPartyDisplay {
  update()
  display()
}
